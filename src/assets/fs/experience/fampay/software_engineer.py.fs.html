# Classes
class Org:
    def __init__(self, name, url, location, description):
        self.name = name; self.url = url; self.location = location
        self.description = description

    def display(self):
        print(f'{self.name} ({self.url}) - {self.location}')
        print(self.description)


class Role:
    def __init__(self, org, title, period, achievements):
        self.org = org
        self.title = title; self.period = period
        self.achievements = '\n'.join(map(lambda ach: f'â€¢ {ach}', achievements))

    def display(self):
        self.org.display()
        print('')
        print(self.title)
        print(f'Period: {self.period}')
        print(f'Achievements:\n{self.achievements}')
