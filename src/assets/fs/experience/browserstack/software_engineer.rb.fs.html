# Classes
class Org
  def initialize(name, url, location, description)
    @name, @url, @location, @description = name, url, location, description
  end

  def display
    puts "#{@name} (#{@url}) - #{@location}"
    puts @description
  end
end

class Role
  def initialize(org, title, period, achievements)
    @org, @title, @period, @achievements = org, title, period, achievements
  end

  def display
    @org.display; puts

    puts "#{@title} - #{@period}"
    puts @achievements.map{ |ach| "â€¢ #{ach}" }.join("\n")
  end
end
