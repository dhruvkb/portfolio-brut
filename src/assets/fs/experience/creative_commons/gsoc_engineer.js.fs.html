// Classes
class Org {
  constructor(name, url, location, description) {
    this.name = name; this.url = url; this.location = location;
    this.description = description;
  }

  display() {
    console.log(`${this.name} (${this.url}) - ${this.location}`);
    console.log(this.description);
  }
}

  constructor(org, title, period, achievements) {
    this.org = org;
    this.title = title; this.period = period;
    this.achievements = achievements.map(ach => `â€¢ ${ach}`).join('\n');
  }

  display() {
    this.org.display();
    console.log('');
    console.log(this.title)
    console.log(`Period: ${this.period}`);
    console.log(`Achievements:\n${this.achievements}`);
  }
}
