// Classes
class Org {
  constructor(name, url, location, description) {
    this.name = name; this.url = url; this.location = location;
    this.description = description;
  }

  display() {
    console.log(`${this.name} (${this.url}) - ${this.location}`);
    console.log(this.description);
  }
}
