// Classes
class Org {
  private String name, url, location, description;

  public Org(String name, String url, String location, String description) {
    this.name = name; this.url = url; this.location = location;
    this.description = description;
  }

  public void display() {
    System.out.println(String.format("%s (%s) - %s", name, url, location));
    System.out.println(description);
  }
}

class Role {
  private Org org;
  private String title, period;
  private String[] achievements;

  public Role(Org org, String title, String period, String[] achievements) {
    this.org = org;
    this.title = title; this.period = period;
    this.achievements = achievements;
  }

  public void display() {
    org.display(); System.out.println('');

    System.out.println(String.format("%s: %s", title, period));
    for (String ach: achievements) { System.out.println("â€¢ " + ach); }
  }
}
